// Simple token transfer with compile-time safety
pub fn transfer(ctx: *Context, amount: u64) !void {
    // Verify sender has sufficient balance at compile time
    if (amount > ctx.accounts.sender.lamports) {
        return error.InsufficientFunds;
    }
    
    // Zero-copy: mutate in place
    ctx.accounts.sender.lamports -= amount;
    ctx.accounts.receiver.lamports += amount;
    
    std.log.info("Transferred {} lamports", .{amount});
}
